# px4_add_module is a CMake function defined in cmake/common/px4_base.cmake
# this function generates commands to build a static library from module desc. 
# input for MODULE specifies the target name of this module
# input for MAIN specifies the entry point of the module
# input for DEPENDS is a string containing the targets this module is based on.
px4_add_module(
        MODULE drivers__sonar_group
        MAIN sonar_group
	SRCS
		sonar_group.c
        DEPENDS
                platforms__common
        )
